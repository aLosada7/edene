## Import

```bash
import { Select } from '@edene/components';
```

## Usage

```tsx
<Select label="Favorite framework" defaultValue="Angular">
    <Option value="React">
        <span>React</span>
    </Option>
    <Option value="Angular">
        <span>Angular</span>
    </Option>
    <Option value="Vue">
        <span>Vue</span>
    </Option>
</Select>
```

### Controlled select

```tsx
() => {
    const [value, setValue] = React.useState('React');
    return (
        <Select label="Favorite framework" value={value} onChange={setValue}>
            <Option value="React">
                <span>React</span>
            </Option>
            <Option value="Angular">
                <span>Angular</span>
            </Option>
            <Option value="Vue">
                <span>Vue</span>
            </Option>
        </Select>
    );
};
```
