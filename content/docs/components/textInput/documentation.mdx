## Import

```bash
import { TextInput } from '@edene/components';
```

## Usage

```tsx
<TextInput label="Your email" placeholder="Your email" />
```

## Controlled

```tsx
() => {
    const [value, setValue] = React.useState('');
    return (
        <TextInput
            label="Your email"
            placeholder="Your email"
            value={value}
            onChange={(event) => setValue(event.target.value)}
        />
    );
};
```

### Error

```tsx
<TextInput label="Your email" placeholder="Your email" error="Invalid email" />
```

### Success

```tsx
<TextInput
    label="Your email"
    placeholder="Your email"
    success="Correct email"
/>
```

### Disabled state

```tsx
<TextInput label="Your email" placeholder="Your email" disabled />
```

### With Icon left

```tsx
<TextInput label="Your email" placeholder="Your email" icon="search" />
```
