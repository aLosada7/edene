## Import

```bash
import { Container, Row, Col } from '@edene/components';
```

## Usage

```tsx
() => {
    const col = css`
        padding-top: 0.75rem;
        padding-bottom: 0.75rem;
        background-color: rgba(86, 61, 124, 0.15);
        border: 1px solid rgba(86, 61, 124, 0.2);
    `;

    return (
        <Container>
            <Row>
                <Col css={col} md={12}>
                    One of two columns
                </Col>
                <Col css={col} md={12}>
                    One of two columns
                </Col>
            </Row>
            <Row>
                <Col css={col} lg={8}>
                    One of three columns
                </Col>
                <Col css={col} lg={8}>
                    One of three columns
                </Col>
                <Col css={col} lg={8}>
                    One of three columns
                </Col>
            </Row>
        </Container>
    );
};
```
