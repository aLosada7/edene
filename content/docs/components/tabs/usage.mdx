## Import

```bash
import { Tabs, Tab } from '@edene/components';
```

## Usage

A default tab.

```tsx
() => {
    const [selected, setSelected] = React.useState('1');

    return (
        <Tabs active={selected} onTabChange={(tabKey) => setSelected(tabKey)}>
            <Tab tabKey="1" label="Tab 1">
                <Text>Content for first tab goes here.</Text>
            </Tab>
            <Tab tabKey="2" label="Tab 2">
                <Text>Content for second tab goes here.</Text>
            </Tab>
        </Tabs>
    );
};
```
